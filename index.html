        <strong>How this works:</strong><br>

        • Starts with the GAO’s upper estimate of <span class="highlight">$521 billion</span> lost annually to federal government fraud (2026 data).<br>

        • Real-time subtraction based on the ongoing Iran war (“Operation Epic Fury”) at ~$1 billion per day (~$11,574 per second), consistent with Pentagon briefings ($11.3 billion in the first 6 days).<br>

        • Numbers update live every second. War cost is simulated from an approximate current total (~$15B+ as of mid-March 2026).<br>

    <script>

        // ==================== CUSTOMIZE THESE VALUES ====================

        let fraud = 521000000000;           // Starting fraud (GAO high: $521 billion)

        let warCost = 15234567890;          // Current estimated Iran war total so far

        const warRatePerSecond = 1000000000 / 86400;  // $1 billion per day

        // =================================================================


        const fraudEl = document.getElementById('fraud');

        const warEl = document.getElementById('war');

        const netEl = document.getElementById('net');


        function formatMoney(amount) {

            return '$' + Math.floor(amount).toLocaleString('en-US');

        }


        function updateDisplay() {

            warCost += warRatePerSecond;

            const net = Math.max(0, fraud - warCost);


            fraudEl.textContent = formatMoney(fraud);

            warEl.textContent = formatMoney(warCost);

            netEl.textContent = formatMoney(net);

        }


        // Live update every second

        setInterval(updateDisplay, 1000);


        // Initial display

        updateDisplay();

    </script>
